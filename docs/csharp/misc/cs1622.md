 <span data-ttu-id="b6c4f-107">Následující ukázka generuje CS1622:</span><span class="sxs-lookup"><span data-stu-id="b6c4f-107">The following sample generates CS1622:</span></span>

class C : IEnumerable

   public IEnumerator GetEnumerator()
