## <a name="example"></a><span data-ttu-id="2ad94-109">Příklad</span><span class="sxs-lookup"><span data-stu-id="2ad94-109">Example</span></span>

// cs1947.cs  
using System.Linq;  
class Test  
{  
    static void Main()  
    {  
        int[] array = new int[] { 1, 2, 3, 4, 5 };
