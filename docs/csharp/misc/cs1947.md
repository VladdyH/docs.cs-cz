## <a name="example"></a><span data-ttu-id="d5944-109">Příklad</span><span class="sxs-lookup"><span data-stu-id="d5944-109">Example</span></span>

class Test  
{  
    static void Main()  
    {  
        int[] array = new int[] { 1, 2, 3, 4, 5 };
