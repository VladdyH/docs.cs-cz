```csharp

{  
   unsafe public void M()  
   {  
      int a = 0, b = 0;  
      int *i = &(a + b);   // CS0211, the addition of two local variables

      // int *i = &a;  
   }  
  
   public static void Main()  
   {  
   }  
}  
```
