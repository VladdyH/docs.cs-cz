 <span data-ttu-id="f98bb-105">Kompilátor jazyka C# neumožňuje použít <xref:System.ParamArrayAttribute?displayProperty=nameWithType>; použít [parametry](../../csharp/language-reference/keywords/params.md) místo.</span><span class="sxs-lookup"><span data-stu-id="f98bb-105">The C# compiler does not allow for the use of <xref:System.ParamArrayAttribute?displayProperty=nameWithType>; use [params](../../csharp/language-reference/keywords/params.md) instead.</span></span>  
  
 <span data-ttu-id="f98bb-106">Následující ukázka generuje CS0674:</span><span class="sxs-lookup"><span data-stu-id="f98bb-106">The following sample generates CS0674:</span></span>  
  
```csharp  
// CS0674.cs  
using System;  
public class MyClass   
{  
  
   public static void UseParams([ParamArray] int[] list)   // CS0674  
   // try the following line instead  
   // public static void UseParams(params int[] list)   
   {  
      for ( int i = 0 ; i < list.Length ; i++ )  
         Console.WriteLine(list[i]);  
      Console.WriteLine();  
   }  
  
   public static void Main()   
   {  
      UseParams(1, 2, 3);  
   }  
}  
```
