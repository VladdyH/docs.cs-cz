## <a name="example"></a><span data-ttu-id="46d4b-106">Příklad</span><span class="sxs-lookup"><span data-stu-id="46d4b-106">Example</span></span>

unsafe class C  
{  
   fixed int a[10];   // CS1642

   public int[] a = null;
