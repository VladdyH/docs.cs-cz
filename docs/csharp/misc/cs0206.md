```csharp  
// CS0206.cs  
public class MyClass

    public static int P  
    {  
        get  
        {  
            return 0;  
        }  
        set

        }  
    }

    }

    {  
        MyMeth(ref P);   // CS0206  
        // try the following line instead  
        // MyMeth(P);   // CS0206  
    }  
}  
```
