## <a name="example"></a><span data-ttu-id="80d67-107">Příklad</span><span class="sxs-lookup"><span data-stu-id="80d67-107">Example</span></span>

using System.Linq;  
  
public class C  
{  
    public static int Main()  
    {  
        int[] array1 = { 1, 2, 3 ,4, 5, 6,};

    }
