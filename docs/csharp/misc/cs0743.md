## <a name="example"></a><span data-ttu-id="fae96-107">Příklad</span><span class="sxs-lookup"><span data-stu-id="fae96-107">Example</span></span>

using System.Linq;  
  
public class C  
{  
    public static int Main()  
    {  
        int[] array1 = { 1, 2, 3 ,4, 5, 6,};  
        int[] array2 = { 5, 6, 7, 8, 9 };

    }  
}
