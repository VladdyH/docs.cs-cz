## <a name="example"></a><span data-ttu-id="5e268-112">Příklad</span><span class="sxs-lookup"><span data-stu-id="5e268-112">Example</span></span>  
 <span data-ttu-id="5e268-113">Následující kód generuje CS0473:</span><span class="sxs-lookup"><span data-stu-id="5e268-113">The following code generates CS0473:</span></span>  
  
```csharp  
// cs0473.cs  
public interface ITest<T>  
{  
    int TestMethod(int i);  
    int TestMethod(T i);  
}  
  
public class ImplementingClass : ITest<int>  
{  
    int ITest<int>.TestMethod(int i) // CS0473  
    {  
        return i + 1;  
    }  
  
    public int TestMethod(int i)  
    {  
        return i - 1;  
    }  
}  
  
class T  
{  
    static int Main()  
    {  
        ImplementingClass a = new ImplementingClass();  
        if (a.TestMethod(0) != -1)  
            return -1;  
  
        ITest<int> i_a = a;  
        System.Console.WriteLine(i_a.TestMethod(0).ToString());  
        if (i_a.TestMethod(0) != 1)  
            return -1;  
  
        return 0;  
    }  
}  
```
