## <a name="example"></a><span data-ttu-id="d496e-107">Příklad</span><span class="sxs-lookup"><span data-stu-id="d496e-107">Example</span></span>

using System.Linq.Expressions;  
  
class Test  
{  
    public static int M(__arglist)

        return 1;  
    }

        Expression<Func<int, int>> f = x => Test.M(__arglist(x)); // CS1952
