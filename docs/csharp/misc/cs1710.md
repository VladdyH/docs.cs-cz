## <a name="example"></a><span data-ttu-id="b1f3c-105">Příklad</span><span class="sxs-lookup"><span data-stu-id="b1f3c-105">Example</span></span>  
 <span data-ttu-id="b1f3c-106">Následující kód způsobí, že se objeví upozornění CS1710.</span><span class="sxs-lookup"><span data-stu-id="b1f3c-106">The following code will cause warning CS1710 to appear.</span></span>  
  
```csharp  
// CS1710.cs  
// compile with: /doc:cs1710.xml  
// To resolve this warning, delete one of the duplicate <typeparam>'s.  
using System;  
class Stack<ItemType>  
{  
}  
  
/// <typeparam name="MyType">can be an int</typeparam>  
/// <typeparam name="MyType">can be an int</typeparam>  
class MyStackWrapper<MyType>  
{  
    // Open constructed type Stack<MyType>.  
    Stack<MyType> stack;  
    public MyStackWrapper(Stack<MyType> s)  
    {  
        stack = s;  
    }  
}  
  
class CMain  
{  
    public static void Main()  
    {  
        // Closed constructed type Stack<int>.  
        Stack<int> stackInt = new Stack<int>();  
        MyStackWrapper<int> MyStackWrapperInt =  
            new MyStackWrapper<int>(stackInt);  
    }  
}  
```
